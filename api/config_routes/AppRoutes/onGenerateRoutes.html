static Route onGenerateRoutes(RouteSettings settings)  {
  switch (settings.name) {
    case '/':
      return _materialRoute(Auth());
    case '/app':
      return _materialRoute(App());
      case '/boards':
      // return _materialRoute(const Boards());
    default:
      return _materialRoute(Auth());
  }
}
